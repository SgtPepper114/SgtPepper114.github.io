#include<bits/stdc++.h>
using namespace std;

int main()
{
ios::sync_with_stdio(false);
int q;
cin >> q;
while(q–)
{
int n;
long long k;
cin >> n >> k;
string p;
cin >> p;
int step = 0;
for(int i = 0; i < n; i++)
{
if(p[i] == ‘0’)
{
if(i > step)
{
if(k >= i-step)
{
p[step] = ‘0’;
p[i] = ‘1’;
k -= i-step;
}
else
{
p[i-k] = ‘0’;
p[i] = ‘1’;
k = 0;
}
if(k == 0) break;
}
step++;
}
}
cout << p << endl;
}
return 0;
}
